async function installDependencies(
  app: App,
  dependencies: Dependency[],
  onProgress: (progress: InstallProgress) => void
): Promise<void> {
  const missingDeps = dependencies.filter(d => !d.installed)

  const progress: InstallProgress = {
    current: 0,
    total: missingDeps.length,
    currentItem: '',
    status: 'installing',
    log: []
  }

  onProgress(progress)

  for (const dep of missingDeps) {
    progress.currentItem = dep.name
    onProgress(progress)

    try {
      // ä¸‹è½½ä¾èµ–
      progress.log.push({
        type: 'info',
        message: `æ­£åœ¨ä¸‹è½½ ${dep.name}...`
      })
      onProgress(progress)

      const packagePath = await marketplace.download(dep.id, dep.version)

      // å®‰è£…ä¾èµ–
      progress.log.push({
        type: 'info',
        message: `æ­£åœ¨å®‰è£… ${dep.name}...`
      })
      onProgress(progress)

      await installDependency(dep.type, packagePath)

      progress.log.push({
        type: 'success',
        message: `${dep.name} å®‰è£…æˆåŠŸ`
      })
      progress.current++
      onProgress(progress)
    } catch (error) {
      progress.log.push({
        type: 'error',
        message: `${dep.name} å®‰è£…å¤±è´¥: ${error.message}`
      })
      progress.status = 'failed'
      onProgress(progress)
      throw error
    }
  }

  progress.status = 'completed'
  onProgress(progress)
}

async function installDependency(type: string, packagePath: string): Promise<void> {
  switch (type) {
    case 'ui':
    case 'model':
    case 'integration':
      await pluginInstaller.install(packagePath)
      break
    case 'skill':
      await skillInstaller.install(packagePath)
      break
    case 'workflow':
      await workflowInstaller.install(packagePath)

  }
}
